# swap-2-var
python program to swap 2 numbers
num1 = 10
num2 = 20
print("defaults\nnum1=", num1, "num2=", num2)             #printing initial values
num1 += num2                                              #num1=30
num2 = num1 - num2                                        #num2=10
num1 -= num2                                              #num1 =20
print("after swap\nnum1=", num1, "num2=", num2)           #print after swapping
